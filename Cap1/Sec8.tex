\section*{Exercício 1.8}
\begin{justify}
Qual é a potência média do sinal periódico de tempo discreto mostrado na Figura 1.15?
\begin{figure}[H]
       \centering
       \includegraphics[width=\textwidth,height=\textheight,keepaspectratio]{./Cap1/Figuras/fig115.png}
          \begin{center}
            Figura 1.15 do livro.
        \end{center}
\end{figure}

    \emph{\textbf{Resposta:}} A potência média associada ao sinal periódico discreto é dada por
        \begin{align*}
            P &= \frac{1}{N}\sum_{n=-N/2}^{N/2}x^2[n]\\
            P &= \frac{1}{8} \left( \sum_{n=0}^{3}(1)^2 + \sum_{n=4}^{7}(-1)^2 \right)\\
            P &= \frac{1}{8} \left( 4 + 4 \right)\\
            P &= 1
        \end{align*} 
\end{justify}
