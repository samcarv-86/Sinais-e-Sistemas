\section*{Exercício 1.14}
\begin{justify}
Encontre as menores frequências angulares para as quais os sinais senoidais de tempo discreto com seguintes períodos fundamentais são periódicos: (a) $N=8$; (b) $N=32$; (c) $N=64$; (d) $N=128$.\\

    \emph{\textbf{Resposta: }}A menor frequência se dará sempre com $m=1$.\\
    
    (a)
    \begin{align*}
        \Omega&=\frac{2\pi}{N}\\
        \Omega&=\frac{2\pi}{8}\\
        \Omega&=\frac{\pi}{4} \text{ rad}
    \end{align*}
    
    (b)
    \begin{align*}
        \Omega&=\frac{2\pi}{N}\\
        \Omega&=\frac{2\pi}{32}\\
        \Omega&=\frac{\pi}{16} \text{ rad}
    \end{align*}
    
    (c)
    \begin{align*}
        \Omega&=\frac{2\pi}{N}\\
        \Omega&=\frac{2\pi}{64}\\
        \Omega&=\frac{\pi}{32} \text{ rad}
    \end{align*}
    
    (d)
    \begin{align*}
        \Omega&=\frac{2\pi}{N}\\
        \Omega&=\frac{2\pi}{128}\\
        \Omega&=\frac{\pi}{64} \text{ rad}
    \end{align*}
\end{justify}
