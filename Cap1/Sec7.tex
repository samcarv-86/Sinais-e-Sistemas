\section*{Exercício 1.7}
\begin{justify}
Qual é a energia total do sinal de tempo discreto mostrado na Figura 1.16?
\begin{figure}[H]
       \centering
       \includegraphics[width=\textwidth,height=\textheight,keepaspectratio]{./Cap1/Figuras/fig116.png}
          \begin{center}
            Figura 1.16 do livro.
        \end{center}
\end{figure}

    \emph{\textbf{Resposta:}} A energia do pulso discreto é dada por
        \begin{align*}
            E &= \sum_{n=-\infty}^{\infty}x^2[n]\\
            E &= \sum_{n=-1}^{1}(1)^2\\
            E &= 3
        \end{align*}    
\end{justify}
